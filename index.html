        async function fetchWeather() {

            try {

                // Replace with actual weather API URL and key

                const response = await fetch('https://api.example.com/weather?location=MEL');

                const data = await response.json();

                document.getElementById('weather-data').innerText = `Temperature: ${data.temperature} °C, Condition: ${data.condition}`;

            } catch (error) {

                document.getElementById('weather-data').innerText = 'Unable to fetch weather data';

            }

        }


        async function fetchFlights() {

            try {

                // Replace with actual flight data API URL

                const response = await fetch('https://api.example.com/flights?airport=MEL');

                const data = await response.json();

                

                const arrivals = data.arrivals;

                const departures = data.departures;


                const arrivalsTable = document.getElementById('arrivals-table');

                const departuresTable = document.getElementById('departures-table');


                arrivalsTable.innerHTML = '';

                departuresTable.innerHTML = '';


                arrivals.forEach(flight => {

                    arrivalsTable.innerHTML += `

                        <tr>

                            <td>${flight.airline}</td>

                            <td>${flight.flight}</td>

                            <td>${flight.sta}</td>

                            <td>${flight.ata || 'N/A'}</td>

                            <td>${flight.arrivalGate || 'N/A'}</td>

                            <td>${flight.aircraft}</td>

                            <td>${flight.delay || 'On Time'}</td>

                        </tr>`;

                });


                departures.forEach(flight => {

                    departuresTable.innerHTML += `

                        <tr>

                            <td>${flight.airline}</td>

                            <td>${flight.flight}</td>

                            <td>${flight.std}</td>

                            <td>${flight.atd || 'N/A'}</td>

                            <td>${flight.departureGate || 'N/A'}</td>

                            <td>${flight.aircraft}</td>

                            <td>${flight.delay || 'On Time'}</td>

                        </tr>`;

                });

            } catch (error) {

                console.error('Error fetching flight data:', error);

            }

        }


        fetchWeather();

        fetchFlights();
